"use client";

import { Button } from "@/components/ui/button";
import {
  Card,
  CardContent,
  CardFooter,
  CardHeader,
  CardTitle,
} from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";
import { Star, ShoppingCart } from "lucide-react";
import Image from "next/image";
import useProductStore from "@/utils/zustand";
import { useEffect } from "react";
import Link from "next/link";

export function ProductCardComponent() {
  const { products, error, isLoading, fetchProducts } = useProductStore();

  const rating = 4;
  const category = "car";
  useEffect(() => {
    fetchProducts();
  }, []);

  if (isLoading) return <p>loading...</p>;
  if (error) return <p>{error}</p>;

  return (
    <div className="flex justify-center flex-wrap gap-4 px-15 py-10">
      {products.map((product) => {
        const { id, name, price, description, imageUrl } = product;
        const productid = id.toString()
        return (
          <Link key={product.id} href={`/products/${productid}`}>
            <Card className="w-full max-w-sm overflow-hidden transition-shadow duration-300 hover:shadow-lg">
              <div className="relative">
                <Image
                  src={imageUrl}
                  alt={name}
                  className="object-cover w-full h-40"
                  width={400}
                  height={400}
                />
                <Badge className="absolute top-2 left-2 bg-primary text-primary-foreground">
                  {category}
                </Badge>
              </div>
              <CardHeader className="p-3 pb-0">
                <CardTitle className="text-base font-semibold line-clamp-1">
                  {name}
                </CardTitle>
              </CardHeader>
              <CardContent className="p-3 pt-2">
                <div className="flex justify-between items-center mb-2">
                  <div className="text-2xl font-bold text-primary">
                    ${price.toFixed(2)}
                  </div>
                  <div className="flex items-center">
                    {[...Array(5)].map((_, i) => (
                      <Star
                        key={i}
                        className={`w-3 h-3 ${
                          i < Math.round(rating)
                            ? "text-yellow-400 fill-yellow-400"
                            : "text-gray-300"
                        }`}
                      />
                    ))}
                    <span className="ml-1 text-xs text-muted-foreground">
                      ({rating.toFixed(1)})
                    </span>
                  </div>
                </div>
                <p className="text-xs text-muted-foreground line-clamp-2 mb-2">
                  {description}
                </p>
              </CardContent>
              <CardFooter className="p-3 pt-0">
                <Button className="w-full text-sm py-1" size="sm">
                  <ShoppingCart className="mr-2 h-3 w-3" /> Add to Cart
                </Button>
              </CardFooter>
            </Card>
          </Link>
        );
      })}
    </div>
  );
}
